{
    if (Modernizr.localstorage)

{

        window.classifiedsearchtool = JSON.stringify(j);

{

    }
    else {
        var isNew = true;
        for (var i=0;i<j.length;i++){

                j[i].val = val;
                isNew = false;
                break;
            }
        }

};
