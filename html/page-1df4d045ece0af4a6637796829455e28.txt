<script type="text/javascript">

$(document).ready(function(){

    $(".saveThisAd").click(function(event) {
        var adId = this.id;
        if(adId.indexOf("gallery") != -1){
            var splitAdId = adId.split("-");
            adId = splitAdId[0];
        }

        // remove from saved ads if class "addIsSaved" is present
        if ($(this).hasClass("adIsSaved"))
        {
            $.ajax({
                type: "POST",
                url: "/donedeal/ajax/removeFromSavedAds",
                data: {ad: adId},
                success: function(response)
                {
                    $('#' + adId).removeClass("adIsSaved");
                    $('#' + adId + "-gallery").removeClass("adIsSaved");
                    $('.qtip-content').html('This ad has been removed from your saved ad\'s');

                    var count = parseInt($('#savedAdCount').text());
                    count = count - 1;
                    $('#savedAdCount').text(count);
                },
                error: function(xhr, status, err) {
                    if (xhr.status == 401) {
                        window.location.href = '/accounts/saved/ads/remove/'+ adId +'?referer' + escape(window.location.href);
                    }
                }
            });
        }
        // otherwise save ad
        else
        {
            $.ajax({
                type: "POST",
                url: "/donedeal/ajax/addToSavedAds",
                data: {ad: adId},
                success: function(data, status, xhr)
                {
                    $('#' + adId).addClass("adIsSaved");
                    $('#' + adId + "-gallery").addClass("adIsSaved");
                    $('.qtip-content').html('This ad has been saved');

                    var count = parseInt($('#savedAdCount').text());
                    count = count + 1;
                    $('#savedAdCount').text(count);
                },
                error: function(xhr, status, err) {
                    if (xhr.status == 401) {
                        window.location.href = '/accounts/saved/ads/add/'+ adId+ '?referer=' + escape(window.location.href);
                    }
                }
            });
        }
        return false;
    });
