    if (numImages % numImagesPerPage > 0) numPages++;
    scroll_next();
    function calculateNumImages() {
        return (jQuery('ul.imgItemContainer').children('li').length);
    }
    function scroll() {
        var startIndex = numImagesPerPage * page;
        var endIndex = numImagesPerPage * (page + 1) - 1;
        var active = false;
        var inactive = false;
        jQuery('ul.imgItemContainer').children('li').each(function(index) {
            if (index < startIndex || index > endIndex) {
                active = jQuery(this).hasClass('Image-item-active');
                if (active) {
                    jQuery(this).removeClass('Image-item-active');
                }
                inactive = jQuery(this).hasClass('Image-item-inactive');
                if (inactive) {
                    jQuery(this).removeClass('Image-item-inactive');
                }
                jQuery(this).addClass('Image-item-inactive');

            }
            else {
                active = jQuery(this).hasClass('Image-item-active');
                if (active) {
                    jQuery(this).removeClass('Image-item-active');
                }
                inactive = jQuery(this).hasClass('Image-item-inactive');
                if (inactive) {
                    jQuery(this).removeClass('Image-item-inactive');
                }
                jQuery(this).addClass('Image-item-active');
            }

        });                
        }
        
    function scroll_next() {
        if (typeof page == "undefined")
            return false;
        if (page >= numPages) {
            page=0;
            scroll();
            return true;
        }
        else {
            page++;
            scroll();
            return true;
        }
       
    }

    function scroll_prev() {
        if (typeof page == "undefined")
            return false;
        if (page <= 0) {
            page=2;
            scroll();
        }
        else {
            page--;
            scroll();
        }
        return false;
    }
