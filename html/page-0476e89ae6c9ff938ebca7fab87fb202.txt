	var fd_cookie = (function() {
		var doc = document,
			undef;
	  
		return {
			// These functions modified from ppk's: http://www.quirksmode.org/js/cookies.html
			set: function (name,value,hours) {
				var expires = "";
				if (hours === undef) {
					this.set(name, value, 10*365);
					return;
				} else if (hours) {
					var date = new Date();
					date.setTime(date.getTime()+(hours*60*60*1000)); //changed to hours instead of days
					expires = "; expires="+date.toGMTString();
				}
				doc.cookie = name+"="+value+expires+"; path=/";
			},
			get: function (name) {
				if(!name) {
					return undef;
				}
			  
				var nameEQ = name + "=",
					cookies = doc.cookie.split(";");
				  
				for(var i=0, l = cookies.length; i < l; i++) {
					var c = cookies[i],
						j = 0,
						clen = c.length;
					
					while (j < clen && c.charAt(j) === ' ') {
						j++;
					}
				
					if(j == clen) {
						continue;
					} else if (c.indexOf(nameEQ,j) === 1) { //Whitespace in beginning of strings, need to trim or move index +1
						return c.substr(nameEQ.length+1);
					}
				}
				return undef;
			},
			clear: function (name) {
				this.set(name,"",-1);
			}
		};
	})();
