$.expr[':'].nottopline = function(elem, index, match){
	var firstsib = $(elem).parent().filter(':first').get(0);
	var firsttop = $(firstsib).offset().top;
	var elemtop = $(elem).offset().top;
	return firsttop != elemtop;

$.expr[':'].topline = function(elem, index, match){
	var firstsib = $(elem).parent().filter(':first').get(0);
	var firsttop = $(firstsib).offset().top;
	var elemtop = $(elem).offset().top;
	return firsttop == elemtop;
