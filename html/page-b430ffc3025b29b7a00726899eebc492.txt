    function valid_email(val) {
        var filter = /[\w-\.]{3,}@([\w-]{2,}\.)*([\w-]{2,}\.)[\w-]{2,4}/;
        if (filter.test(val))
            return true;
        else
            return false;
    }

    function valid_phone(strPhone) {
        var _sTexto = "";
        var numbers = ["000-0000", "555-1212", "222-2222", "333-3333", "444-4444", "555-5555", "666-6666", "777-7777", "888-8888", "999-9999", "0000000", "5551212", "2222222", "3333333", "4444444", "5555555", "6666666", "7777777", "8888888", "9999999"];

        if (strPhone.length == 10) {
            _sTexto = strPhone.substring(3);
        }
        else {
            if (strPhone.length == 12) {
                _sTexto = strPhone.substring(4);
            }
        }

        if ($.inArray(_sTexto, numbers) >= 0) {
            return false;
        }
        else {
            var pattern = "^\\(?([2-9][0-8][0-9])\\)?[-. ]?([2-9][0-9]{2})[-. ]?([0-9]{4})$";
            var myregexp = new RegExp(pattern);

            return myregexp.test(strPhone);
        }
    }

    function valid_ZipUS(strZipCode) {
        var badZip = ["00000"];

        if (strZipCode.length > 5) {
            return false;
        }

        if ($.inArray(strZipCode, badZip) >= 0) {
            return false;
        }
        else {
            var pattern = "^\\d{5}(-\\d{4})?$";
            var myregexp = new RegExp(pattern);

            return myregexp.test(strZipCode);
        }
    }

    function valid_ZipCA(strZipCode) {
        var badZip = [""];

        if (strZipCode.length > 6) {
            return false;
        }

        if ($.inArray(strZipCode, badZip) >= 0) {
            return false;
        }
        else {            
            var pattern = "^[ABCEGHJKLMNPRSTVXY]{1}\\d{1}[A-Z]{1} *\\d{1}[A-Z]{1}\\d{1}$";
            var myregexp = new RegExp(pattern);

            return myregexp.test(strZipCode);
        }
    }
