function restructure_sub_nav(li_id){
        var sub_width=0;
        var parent_width=$("#"+li_id).innerWidth();
        var nav_width=$("#"+li_id).parent().innerWidth();
        $("#"+li_id+" > div.subnav > ul.sub_col").each(function(i){
                sub_width+=$(this).innerWidth();
//              alert(sub_width);
        });
        if (sub_width < parent_width){
                sub_width = parent_width;
        }
		
		if (sub_width > nav_width)
		{
			remove_sub_nav_column(li_id);
			sub_width=restructure_sub_nav(li_id);
		}
		
        $("#"+li_id+" > div.subnav").width(sub_width+2);
        return sub_width;
}

			var rows=min_rows;

<script>
/*
	S. PARK 
  	Menu slide-out code
	====================================
	price_global_id holds the setTimeout ID of the 'Price' slide-out menu.
	make_global_id holds the setTimeout ID of the 'Makes' slide-out menu.
	
	I use setTimeout to close the slide menu's upon leaving the link area.
	I give the user enough time to move the mouse over to the slide menu before it closes.
	Note that I use both mouseleave and mouseout events.
*/
$(document).ready(function(){
	var price_global_id = null;
	var make_global_id = null;
	
	$('#search_by-price').mouseover(function(){
		if (price_global_id != null)
		{
			clearTimeout(price_global_id);
			price_global_id = null;
		}
		
		if ($('#menu_slider-makes').css('display') === 'block')
			$('#menu_slider-makes').css('display', 'none');
		
		$('#menu_slider-makes').stop(true, true);
		$('#menu_slider-price').show('fast');
	});
	
	$('#search_by-make').mouseover(function(){
		if (make_global_id != null)
		{
			clearTimeout(make_global_id);
			make_global_id = null;
		}
		
		if ($('#menu_slider-price').css('display') === 'block')
			$('#menu_slider-price').css('display', 'none');
			
		$('#menu_slider-price').stop(true, true);
		$('#menu_slider-makes').show('fast');
	});
	
	$('#search_by-make').mouseout(function(){
		make_global_id = setTimeout(function(){
		if ($('#menu_slider-makes').css('display') === 'block' && !$('#menu_slider-makes').is(':hover'))
			$('#menu_slider-makes').css('display', 'none');
		}, 400);
	});
	
	$('#search_by-price').mouseout(function(){
		price_global_id = setTimeout(function(){
		if ($('#menu_slider-price').css('display') === 'block' && !$('#menu_slider-price').is(':hover'))
			$('#menu_slider-price').css('display', 'none');
		}, 400);
	});
	
	$('#menu_slider-makes').mouseleave(function(){
		$(this).css('display', 'none');
	});
	
	$('#menu_slider-price').mouseleave(function(){
		$(this).css('display', 'none');
	});
});
</script>
