        __squid_like.member_likes ? dec() : inc();

              $Sq.notify.error("Hey there! You've reached the maximum number of SquidLikes for today. Come back again tomorrow to SquidLike this lens.", { hideAfter: 5000 });

          },
          'text', 
          function() { // error
            previousState ? inc() : dec();
            allButtons.removeClass('disabled');
          }
        );
      }

      // this only happens if the Squidoo.login hasn't executed
