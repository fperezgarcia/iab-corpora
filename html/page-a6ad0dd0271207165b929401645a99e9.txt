					function selectInfo(sel)
					{
					
						var n = 0;
						var s = [];
						for(n=0;n<sel.options.length;n++)
							s[n] = {name:sel.options[n].text,value:sel.options[n].value,selected:sel.options[n].selected};
						return s;
					}
					
					function selectLoad(w,d)
					{
					
						if(!w) return false;
						if(!d || d.length<=1) return false;
						
						var old_data = selectInfo(w);
						
						for(var c=w.options.length+1;c>=0;--c)
							w.options[c]=null;
						
						d.each(function(e){w.options[w.options.length] = new Option(e.name,e.name);});
						
						for(var n=0;n<old_data.length;n++)
						{
							if(old_data[n].selected)
								if(_opt_obj = w.getElements("option[value=" + old_data[n].value + "]")[0])
									_opt_obj.selected = true;
						}
						return false;
					}
					
					function compareOptionText(a,b)
					{  
					
						if(a.text == "All") return -1;
						if(b.text == "All") return 1;
							
						return a.value!=b.value ? a.value<b.value ? -1 : 1 : 0;
						// return a.text!=b.text ? a.text<b.text ? -1 : 1 : 0;
						// numerical comparison
						// return a.text - b.text; 
					}
	
					function sortOptions(list)
					{
					
						var items = list.options.length;
						var tmpArray = new Array(items);
						var old_data = selectInfo(list);
						
						for ( i=0; i<items; i++ )
							tmpArray[i] = new Option(list.options[i].text,list.options[i].value,list.options[i].selected);  
						
						tmpArray.sort(compareOptionText);  
						  
						for ( i=0; i<items; i++ )
							list.options[i] = new Option(tmpArray[i].text,tmpArray[i].value,false);
						
						for(var n=0;n<old_data.length;n++)
						{
							if(old_data[n].selected)
								if(_opt_obj = list.getElements("option[value=" + old_data[n].value + "]")[0])
									_opt_obj.selected = true;
						}
						return false;
					}
					
					function selectResetOrder(selObj,data)
					{
					
						if(!selObj) return false;
						
						selname = selObj.name;

						{
							$(selname+'_st_a1').style.textDecoration='none';
							$(selname+'_st_a2').style.textDecoration='';
							sortOptions($(selname));
						}
						return false;
					}
