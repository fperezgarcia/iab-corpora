    var rbw = new function() {

      this.start = function() {
        this.results = {};
        for (i in this.partner_ranking) {
          this.start_data(this.partner_ranking[i]);
        }
        var self = this;
        setTimeout(function() {
          self.handle_results();
        }, 750); // We're moving on whether or not we got responses back by then.
      };

      this.start_data = function(partner_name) {
        var api_url = "http://api.merchantcircle.com/api/v1/company/get_related_businesses_" + partner_name + "?id=8420334&callback=rbw_" + partner_name + "_callback";

        var widget = jQuery('#alsoViewedContent');

          if (partner_name in this.results) {

              comps.push(this.results[partner_name][i]);
            }
          }
        }

        comps = comps.slice(0, 3); // First three results

        if (comps.length > 0) {
          for (i in comps) {
            var comp = comps[i];

          pwb.append(comp.impression_pixel);
        }

        pwb.click(function () {
          if (!jQuery(this).hasClass('clicked')) {
            jQuery(this).addClass('clicked');

        }
