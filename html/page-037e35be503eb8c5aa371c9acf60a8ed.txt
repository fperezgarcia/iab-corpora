	function submitTravelForm (tabForm) {
		var isVal = true;
		
		$(tabForm).find('[data-req=true]').each( function(i, el) {
			var _input = $(this);
			
			switch (el.type) {

					_input.blur( function() {
						if ($(this).val() != "") {
							$(this).removeClass('error');
						}
					});

					_input.closest('select').addClass('error');
					_input.closest('select').change( function() {
						if ($(this).find('option:selected').val() != "") {
							$(this).removeClass('error');
						}
