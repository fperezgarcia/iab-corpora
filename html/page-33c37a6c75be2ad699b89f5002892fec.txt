function Popup(inOptions)
{
    var self = this;
    var itsMousePosition = {left: 0, top: 0};
    var itsMessageBody;
    var mouseInside = false;
    var theReplacementContainer;
    var itsOptions = {
        fadeInterval: 400,
        delay: 1000,
        containerId: "#pop-up",
        bodyContainerId: "#pop-up-message-body",
        closeButtonContainerId: "pop-up-close",
        vAlign: "bottom",
        hAlign: "left",
        leftIndent: 15,
        topIndent: 15
    };
    itsOptions = jQuery.extend({}, itsOptions, inOptions);
    self.setMousePosition = function(inLeft, inTop)
    {
        itsMousePosition.left = inLeft;
        itsMousePosition.top = inTop;
    };
    self.setMessageBody = function(inMessage)
    {
        itsMessageBody = inMessage;
        theReplacementContainer = null;
    };
    self.setContainer = function(inContainerId)
    {
        theReplacementContainer = inContainerId;
        itsMessageBody = null;
    };
    self.isMouseInside = function(isMouseInside)
    {
        mouseInside = isMouseInside;
    };
    self.hide = function(inEvent)
    {
        mouseInside = false;
        jQuery(itsOptions.containerId).fadeOut(400);
    };
    self.display = function(inDelay, inEvent)
    {
        var theDelay = inDelay ? inDelay :itsOptions.delay;
        var left = itsMousePosition.left;
        var top = itsMousePosition.top;
        var thePopupContainer = theReplacementContainer == null ? itsOptions.containerId : theReplacementContainer;
        var theReplacementContainerContent = jQuery(thePopupContainer).html();
        jQuery(itsOptions.containerId).html(theReplacementContainerContent);
        var theClassAttribute = jQuery(thePopupContainer).attr("class");
        jQuery(itsOptions.containerId).attr("class", theClassAttribute);
        if(theReplacementContainer == null && itsMessageBody)
            jQuery(itsOptions.bodyContainerId).html(itsMessageBody);
        if(itsOptions.vAlign == "top")
            top = top - jQuery(itsOptions.containerId).height();
        else
            top = top + itsOptions.topIndent;
        if(itsOptions.hAlign == "left")
            left = left - jQuery(itsOptions.containerId).width() + itsOptions.leftIndent;
        else
            left = left - itsOptions.leftIndent;
        jQuery(itsOptions.containerId).css("left", left + "px");
        jQuery(itsOptions.containerId).css("top", top + "px");
        if(itsOptions.closeButtonContainerId)
        {
            jQuery(itsOptions.closeButtonContainerId).click(function() { jQuery(itsOptions.containerId).hide();	});
        }
        else
        {
            jQuery(itsOptions.containerId).mouseover(function(event) { jQuery(this).show(); });
            jQuery(itsOptions.containerId).mouseout(function(event) { jQuery(this).hide();	});
        }
        // Wait for the delay and display popup ONLY IF the user has the mouse over the icon for popup.
        setTimeout(function() {if(mouseInside) jQuery(itsOptions.containerId).fadeIn(400);}, theDelay);
    };
}

function getYearList(){
    var retVal = new Array;
    var currentYear  = new Date().getFullYear();
    for(var i=currentYear+1; i>=1990 ;i--){
        retVal.push(i);
    }
    return retVal;
}

function isValidzipcode(isValid) {
    var errorMessage = new Array();
    var yearValue = jQuery('#year_list').val();
    var makeValue = jQuery('#make_list').val();
    var modelValue = jQuery('#model_list').val();
    var messageIndex = 0;
    jQuery("#errors").hide();
    if(yearValue == -1){
        errorMessage[messageIndex] = "Please select an year";
        messageIndex ++;
    }
    if(makeValue == -1){
        errorMessage[messageIndex] = "Please select a make";
        messageIndex ++;
    }
    if(modelValue == -1){
        errorMessage[messageIndex] = "Please select a model";
        messageIndex ++;
    }
    if(!isValid){
        errorMessage[messageIndex] = "Please enter a valid ZIP code.";
    }
    displayInvalidMessage(errorMessage);
    if(errorMessage.length > 0) {
        return;
    }
    document.repairPalStep2.action = '/cro/cars/repair/service/index.htm';
    var cookieString = "{'year':'" + yearValue + "','make':'" + makeValue + "','ctid':'" + modelValue+ "','zip':'" + jQuery('#zipcode').val() +"' }";
    jQuery.cookie("repair", cookieString, cookieOptions);
    document.repairPalStep2.submit();
}
