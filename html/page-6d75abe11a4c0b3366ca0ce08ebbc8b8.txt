    function updateMessageZC(msg) {
        var message = document.getElementById("valMessageZC");
        message.innerHTML = "<font style='color: #FF0000; font-family: Verdana; font-size: 11px'>" + msg + "</font>";
    }
    
    function checkLengthZC(zipcode, n, min) {
        if (zipcode.length < min) {
            updateMessageZC("Length of " + n + " must be " + min + " digit.");
            return false;
        } else {
            return true;
        }

    }

    function checkRegexpZC(zipcode, regexp, n) {
        if (!(regexp.test(zipcode))) {
            updateMessageZC(n);
            return false;
        } else {
            return true;
        }
    }


    function checkZipcodeZC() {

        var bValid = true;
        var zipCode = $("#txtZipcodeZC").val();

        bValid = bValid && checkLengthZC(zipCode, "Zip Code", 5);
        bValid = bValid && checkRegexpZC(zipCode, /^\d{5}$/, "Please enter a valid 5 digit ZIP code.");
       
        if (bValid) {
            $.post("/Page/VerifySaveZipCode", { zipCode: $("#txtZipcodeZC").val()}, function(data) {
                
                if (data == 'false') {
                    updateMessageZC("Please enter a valid zip code.");
                }
                else {
                    $.post("/Page/ZipCode", 
                        { zipCode: zipCode }, 
                        function(data) { 
                            $('body').trigger('zipSuccess', [zipCode]); //throw custom event so we can do stuff to the page
                            var url = window.location.href;
                            if ($('.changethiszipcode').length > 0) {
                                $('.changethiszipcode').html(zipCode);
                                getnewdealers(zipCode);
                            }
                            if (url.indexOf('#refresh') > -1) { 
                                url = url.replace('#refresh', ''); 
                                window.location.href = url;
                            }
                            NADAjs.AutoTraderWidget.SearchNewZip(zipCode);
                        });
                    
                    $.modal.close();
                }
            });
        }
    }

    function updateMessageCH(msg) {
        var message = document.getElementById("valMessageCH");
        message.innerHTML = "<font style='color: #FF0000; font-family: Verdana; font-size: 11px'>" + msg + "</font>";
    }

    function checkLengthCH(zipcode, n, min) {
        if (zipcode.length < min) {
            updateMessageCH("Length of " + n + " must be " + min + " digit.");
            return false;
        } else {
            return true;
        }

    }

    function checkRegexpCH(zipcode, regexp, n) {
        if (!(regexp.test(zipcode))) {
            updateMessageCH(n);
            return false;
        } else {
            return true;
        }
    }


    function checkZipcodeCH() {

        var bValid = true;
        var zipCode = $("#txtZipcodeCH").val();

        bValid = bValid && checkLengthCH(zipCode, "Zip Code", 5);
        bValid = bValid && checkRegexpCH(zipCode, /^\d{5}$/, "Please enter a valid 5 digit ZIP code.");
        if (bValid) {

            $.post("/Page/VerifySaveZipCode", { zipCode: zipCode }, function(data) {
                if (data == false && $('.changethiszipcode').length > 0) {
                    updateMessageCH("Please enter a valid zip code.");
                }
                else {
                    $.post("/Page/ZipCode",
                    { zipCode: zipCode }, function(data) {
                    var url = window.location.href; if (url.indexOf('#refresh') > -1) { url = url.replace('#refresh', ''); window.location.href = url; }
                    NADAjs.AutoTraderWidget.SearchNewZip(zipCode);

                    if ($('.changethiszipcode').length > 0) {
                        $('.changethiszipcode').html(zipCode);
                        NADADealerQuotejs.DealerQuote.GetNewDealers(zipCode);
                    }
                });

                    $.modal.close();

                }
            });
        }
    }
    
    function CheckCookiesCH() {
        document.cookie = "Enabled=true";
        var cookieValid = document.cookie;
        if (cookieValid.indexOf("Enabled=true") != -1) {
            $("#txtZipcodeCH").val(getCookieCH("zipcode"));
        }
    }

    function getCookieCH(c_name) {
        if (document.cookie.length > 0) {
            c_start = document.cookie.indexOf(c_name + "=");
            if (c_start != -1) {
                c_start = c_start + c_name.length + 1;
                c_end = document.cookie.indexOf(";", c_start);
                if (c_end == -1) c_end = document.cookie.length;
                return unescape(document.cookie.substring(c_start, c_end));
            }
        }
        return "";
    }
