	while (targetURL != targetURL.replace('+',' '))
	{
		targetURL = targetURL.replace('+',' ');
	}	
	targetURL = targetURL.replace('title=','');
	if(null!=targetURL.match('%')){
		var inx1 = targetURL.indexOf('%');
		var strRmv =	targetURL.substring(inx1,inx1+3);
		targetURL = targetURL.replace(strRmv,'');
	}	
	return targetURL;

	function getCookie(name) {
		var dc = document.cookie;

		if (begin == -1) {
			begin = dc.indexOf(prefix);
			if (begin != 0) return null;
		}

		{
			begin += 2;
			var end = document.cookie.indexOf(";", begin);
			if (end == -1) {
			end = dc.length;
			}
		}
		return unescape(dc.substring(begin + prefix.length, end));
