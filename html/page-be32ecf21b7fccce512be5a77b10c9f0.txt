 function isFloat(s){
  	var patrn =/^([1-9]\d*)$|(0)$/;
  	var patrn1=/^([0-9]\d*)$/
  	var dotindex=s.indexOf(".");
  	if(dotindex>0)
  	{
  		var bs=s.substring(0,dotindex);
  		if (!patrn.exec(bs))
  		return false;
  		if(dotindex==s.length-1){
  			return true;
  		}else{
  			var es = s.substring(dotindex+1);
  			if (!patrn1.exec(es))
  			return false;
  			else
  			return true;
  		}
  	}
  	if (!patrn.exec(s)&&s.length<8)
  	return false;
  	else
  	return true;
  }
