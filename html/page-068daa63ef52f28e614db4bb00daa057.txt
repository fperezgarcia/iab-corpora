		var currentProtocol = "";

			currentProtocol = currentURL.substring(0, currentURL.indexOf("://"));

		if (currentProtocol != savedProtocol) {
			absoluteURL = currentProtocol + absoluteURL.substring(absoluteURL.indexOf("://"));

	}

}
