		function showDateTab(tab) {
			var tabs = document.getElementById("events-results").getElementsByTagName("ul");
			for(i = 0; i < tabs.length; i++) {
				if(tabs[i].className.indexOf("day-hide") == -1) {
					tabs[i].className += " day-hide" ;  //just append hiding class, other classes may need to remain
				}
				document.getElementById('day'+i).className="day-unselected"; 
			}
			document.getElementById('event-list-' + tab).className = document.getElementById('event-list-' + tab).className.replace(/day-hide/,'');
			document.getElementById('day'+tab).className="day-selected"; 
		}
		showDateTab(0);
