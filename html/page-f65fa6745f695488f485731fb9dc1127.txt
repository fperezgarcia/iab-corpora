		function GetHttpRequest()
		{			
			if(window.XMLHttpRequest)
			{
				http_request = new XMLHttpRequest();
				if(http_request.overrideMimeType)
				{
					http_request.overrideMimeType("text/xml");
				}
			
				/*if(http_request.setRequestHeader)	
					http_request.setRequestHeader("Content-Type","application/x-www-form-urlencoded");	*/
			}		
			else if(window.ActiveXObject)
			{
				http_request = new ActiveXObject("MicroSoft.XMLHTTP");
			}			
			
			return http_request;	
		}
		
		function MakeRequest(catalog)
		{
			ShowHideLoadingDiv(false);
			ShowHideLoadingDiv1(true);
			var url = "/AjaxHandlers/AjaxResponse.aspx";
			http_request = GetHttpRequest();	
			if(http_request!=null)
			{
				var ddlYear = document.getElementById("CatalogSearch1_ddlYear");
				var yearText = ddlYear.options[ddlYear.selectedIndex].text;
				var yearCode = ddlYear.options[ddlYear.selectedIndex].value;
				url += "?"+"yearText="+yearText+"&yearCode="+yearCode+"&catalog="+catalog;
				http_request.onreadystatechange = FillMake
				http_request.open("GET",url,true);
				http_request.send(null);
			}
		}
		
		function FillMake()
		{
		    
			var ddlMake = document.getElementById("CatalogSearch1_ddlMake");
			for(var j=0; j<=ddlMake.length; j++)
				ddlMake.remove(0); 
			if(ddlMake.length==0)	
				ddlMake.options[ddlMake.length] = new Option("---Select---" , "-1");	
			if(http_request.readyState == 4)
			{
				if(http_request.status == 200)
				{
					var makeXMLDocument = http_request.responseXML; 		
					if(makeXMLDocument!=null)
					{
						var makeCodeList = makeXMLDocument.getElementsByTagName("MakeCode");
						var makeTextList = makeXMLDocument.getElementsByTagName("MakeText");
						if(makeCodeList!=null)
						{
						    
							for(var i=0; i<makeCodeList.length; i++)
							{
								ddlMake.options[ddlMake.length] = new Option(makeTextList.item(i).firstChild.data,makeCodeList.item(i).firstChild.data);	
							}
							if(makeCodeList.length==1)
							{
							ddlMake.options.selectedIndex=1;
							ModelRequest();
							}
							
							ShowHideLoadingDiv(false);
			                ShowHideLoadingDiv1(true);
						}
					}	
				}
					
			}
			
		}

		function ValidateSelection()
		{
			var valid = true;
			var ddlYear= document.getElementById("CatalogSearch1_ddlYear");
			var ddlMake= document.getElementById("CatalogSearch1_ddlMake");
			var ddlModel= document.getElementById("CatalogSearch1_ddlModel");
			var ddlEngine= document.getElementById("CatalogSearch1_ddlEngine");
			var trEngine1 = document.getElementById("CatalogSearch1_trEngine");
			
			if(ddlYear.selectedIndex<=0)
			{
				valid = false;
    		}
			
			if(ddlMake.selectedIndex<=0)
			{
				valid = false;
			}
				
			if(ddlModel.selectedIndex<=0)
			{
				valid = false;
			}
				
			if(trEngine1.style.display!="none")
			{
			if(!(ddlEngine.length<1))
			{	
			    if(ddlEngine.selectedIndex<=0)
				valid = false;	
			}
		    }
			
			if(!valid)
				document.getElementById("CatalogSearch1_spanError").style.display = "block";	
			else
			{
				document.getElementById("CatalogSearch1_spanError").style.display = "none";						
				document.getElementById("CatalogSearch1_hdnYearCode").value = ddlYear.options[ddlYear.selectedIndex].value;
				document.getElementById("CatalogSearch1_hdnMakeCode").value = ddlMake.options[ddlMake.selectedIndex].value;
